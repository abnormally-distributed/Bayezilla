model{


for (i in 1:(N-1)) {
  for (j in (i+1):N) {
    Sigma[i,j] <- sigma2 * exp(- pow(r*d[i,j],2))
    Sigma[j,i] <- Sigma[i,j]
  }
}

for (k in 1:N) {
    Sigma[k,k] <- sigma2 + w
}  
  
w ~ dgamma(0.04 , 0.04)
sigma2 ~  dgamma(4.0, 0.4) 
r ~ dt(0, 1, 1) T(0,)

for (i in 1:N) {
  mu[i] ~ dnorm(0, 1)
}

y ~ dmnorm.vcov(mu[1:N], Sigma[1:N, 1:N])
ySim ~ dmnorm.vcov(mu[1:N], Sigma[1:N, 1:N])

}
